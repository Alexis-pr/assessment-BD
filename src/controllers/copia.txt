    let total = 0

    for (const r of rows) {
      await insertCustomer({
        name: r.customer_name,
        email: r.customer_email,
        phone: String(r.customer_phone ?? ""),
        address: r.customer_address
      });

      total++;
      console.log(` ${total}/${rows.length}`);
